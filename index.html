Django Playground

How This Works

Django Playground runs your python code client side directly in your browser using pyscript. There is no server.

When this website is launched in your browser, it runs some initialization code in your browser to:

  • Create a model 
                class Dog(models.Model):
                name = models.TextField()
  • Configure django to work with an in memory sqlite database in your browser
  • Migrate with syncdb to create the table in the sqlite database
import os from types import SimpleNamespace import django from django.db import models from django.core.management import call_command from django.conf import settings from django.apps import AppConfig from django.utils.crypto import get_random_string settings.configure( DEBUG=True, ALLOWED_HOSTS=["*"], ROOT_URLCONF=__name__, SECRET_KEY=get_random_string(50), INSTALLED_APPS=[f"{__name__}.Config"], DATABASES={ 'default': { 'ENGINE': 'django.db.backends.sqlite3', 'NAME': ':memory:', } }, DEFAULT_AUTO_FIELD='django.db.models.BigAutoField', ) registry = [] class Config(AppConfig): name = __name__ label = __name__ # It doesn't have a path at runtime in the browser path = "nopath" def import_models(self): # Dictionary of models for this app, primarily maintained in the # 'all_models' attribute of the Apps this AppConfig is attached to. self.models = self.apps.all_models[self.label] # This is the actual code that looks for a models_module and loads it # if module_has_submodule(self.module, MODELS_MODULE_NAME): # models_module_name = "%s.%s" % (self.name, MODELS_MODULE_NAME) # self.models_module = import_module(models_module_name) # This is making a namespace that acts like the models_module without to import it class Dog(models.Model): name = models.TextField() def __str__(self): return self.name class Meta: app_label = __name__ models_module = SimpleNamespace(Dog=Dog) self.models_module = models_module registry.append(Dog) django.setup() os.environ["DJANGO_ALLOW_ASYNC_UNSAFE"] = "true" call_command("migrate", run_syncdb=True, verbosity=0) Dog = registry[0] dogs = Dog.objects.bulk_create([ Dog(name='Bruce'), Dog(name='Mitch'), Dog(name='Daisy'), ])

Run whatever you want

The data is only running in memory in your browser. If you destroy the database, refresh the page.
